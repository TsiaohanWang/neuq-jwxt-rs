9/// HTTP 客户端的身份标识字符串
10const USER_AGENT: &str =
11    "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 Chrome/108.0.0.0 Safari/537.36";
12/// HTTP 客户端的全局请求超时
13const TIMEOUT: std::time::Duration = std::time::Duration::from_secs(10);
14/// HTTP 客户端的操作间隔
15const OPERATION_INTERVAL: std::time::Duration = std::time::Duration::from_secs(2);
16
17/// 发送请求的教务系统 URL
18const BASE_URL: &str = "https://jwxt.neuq.edu.cn/eams";
19/// 发送登录请求的教务系统 URL 后缀
20const LOGIN_URL_SUFFIX: &str = "/loginExt.action";
21/// 提取登录 salt 的正则表达式
22const LOGIN_SALT_REGEX: &str = r"CryptoJS\.SHA1\('([^']+)-' \+ form\['password'\]\.value\)";
23/// 检验登录是否成功的重定向 URL 后缀
24const LOGIN_VERIFIED_SUFFIX: &str = "/homeExt.action";
25/// 发送空闲教室查询请求的教务系统 URL 后缀
26const CLASSROOM_URL_SUFFIX: &str = "/classroom/apply/free!search.action";
27
28// const COURSE_TABLE_URL_SUFFIX: &str = "/courseTableForStd!courseTable.action";

30/// 与教务系统进行交互的核心 HTTP 客户端
31#[derive(Debug)]
32pub struct NeuqClient {
33    client: Client,
34    base_url: String,
35}
36
37impl NeuqClient {
38    /// 构建一个 HTTP 客户端
39    /// 
40    /// - 已定义的常量会参与客户端构建。
41    pub fn new() -> anyhow::Result<Self> {
42        let mut headers = header::HeaderMap::new();
43        headers.insert(
44            "X-Requested-With",
45            header::HeaderValue::from_static("XMLHttpRequest"),
46        );
47        headers.insert(
48            "Accept",
49            header::HeaderValue::from_static(
50                "application/json, text/javascript, text/html, */*; q=0.01",
51            ),
52        );
53
54        let client = Client::builder()
55            .cookie_store(true)
56            .user_agent(USER_AGENT)
57            .default_headers(headers)
58            .timeout(TIMEOUT)
59            .build()?;
60
61        info!("NeuqClient initialized.");
62
63        Ok(Self {
64            client,
65            base_url: BASE_URL.to_owned(),
66        })
67    }
68
69    /// 获取教务系统登录页面的 salt
70    /// 
71    /// - 已定义的常量会参与 salt 的获取。
72    pub async fn get_salt(&self) -> anyhow::Result<String> {
73        let login_url = format!("{}{}", self.base_url, LOGIN_URL_SUFFIX);
74        let login_html = self.client.get(&login_url).send().await?.text().await?;
75
76        info!("Login html received.");
77
78        let re = Regex::new(LOGIN_SALT_REGEX)?;
79
80        return if let Some(caps) = re.captures(&login_html) {
81            info!("Login salt captured.");
82            Ok(caps[1].to_string())
83        } else {
84            Err(anyhow::anyhow!("Failed to retrieve salt."))
85        };
86    }
87
88    /// 使用传入的学号 `username` 和密码 `password` 进行登录，
89    /// 返回是否成功登录的 `bool` 值
90    /// 
91    /// - 已定义的常量会定义操作间隔，并参与登录成功的判定。
92    pub async fn login(&self, username: &str, password: &str) -> anyhow::Result<bool> {
93        let salt = self.get_salt().await?;
94
95        tokio::time::sleep(OPERATION_INTERVAL).await;
96
97        let mut hasher = Sha1::new();
98        Digest::update(&mut hasher, format!("{}-{}", salt, password).as_bytes());
99        let hashed_password = hex::encode(hasher.finalize());
100
101        let params = [
102            ("username", username),
103            ("password", &hashed_password),
104            ("session_locale", "zh_CN"),
105        ];
106
107        let login_url = format!("{}{}", self.base_url, LOGIN_URL_SUFFIX);
108        let res = self.client.post(&login_url).form(&params).send().await?;
109
110        info!("Login response received.");
111        debug!("Login response: {:?}", &res);
112
113        Ok(res.url().path().contains(LOGIN_VERIFIED_SUFFIX))
114    }
115
116    /// 向服务端指定 URL 后缀 `url_suffix` 发送 POST 请求，
117    /// 其中请求体实现了 `Query` 特征
118    pub async fn post<Q: Query>(
119        &self,
120        url_suffix: &str,
121        query_params: Q,
122    ) -> anyhow::Result<Response> {
123        let url = format!("{}{}", self.base_url, url_suffix);
124        let params: Vec<(String, String)> = query_params.into();
125
126        let res = self.client.post(&url).form(&params).send().await?;
127
128        Ok(res)
129    }
130
131    /// 向服务端发送空闲教室查询的请求，
132    /// 并返回接收到的 HTML 文档
133    /// 
134    /// - 其中请求体参数由 `ClassroomQuery` 定义。
135    pub async fn get_classroom_html(&self, query_params: ClassroomQuery) -> anyhow::Result<String> {
136        let res = self.post(CLASSROOM_URL_SUFFIX, query_params).await?;
137
138        let html = res.text().await?;
139
140        Ok(html)
141    }
